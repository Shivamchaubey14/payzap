Integrate in under 5 minutes

Add one script tag and two lines of JavaScript. Works on any website — React, Vue, plain HTML, WordPress.

Get your API key →
checkout.html 
<!-- Step 1: Add the script -->
<script src="https://api.payzap.com/static/js/checkout.js"></script>

<!-- Step 2: Open the payment popup -->
<script>
  var options = {
    key: "rzp_live_YOUR_KEY",
    amount: 50000, // ₹500 in paise
    currency: "INR",
    name: "Your Store",
    handler: function(response) {
      // Payment successful!
      console.log(response.payment_id);
    }
  };
  var rzp = new PayZap(options);
  rzp.open();
</script>
